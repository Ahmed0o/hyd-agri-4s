function doGet(e){return handle(e)} function doPost(e){return handle(e)} function handle(e){ try{ var ss=SpreadsheetApp.getActiveSpreadsheet(); var sh=ss.getSheets()[0]; var p=e.parameter,a=p.action,b={}; if(e.postData)b=JSON.parse(e.postData.contents); if(a==="getRoutes"){ var lastCol=sh.getLastColumn(); var lastRow=sh.getLastRow(); var routes=[]; for(var c=2;c<=lastCol;c++){ var routeName=sh.getRange(1,c).getValue()||("خط "+(c-1)); var customers=[]; for(var r=3;r<=lastRow;r++){ var val=sh.getRange(r,c).getValue(); if(val)customers.push({name:String(val),row:r}); } if(customers.length||routeName)routes.push({col:c,name:String(routeName),customers:customers}); } var dateVal=sh.getRange(1,1).getValue(); return out({ok:true,routes:routes,date:String(dateVal||"")}); } if(a==="saveRoute"){ var col=b.col; var customers=b.customers; var lastRow=sh.getLastRow(); // clear old data in this column from row 3 if(lastRow>=3)sh.getRange(3,col,lastRow-2,1).clearContent(); // write new order for(var i=0;i<customers.length;i++){ sh.getRange(3+i,col).setValue(customers[i].name); } return out({ok:true}); } return out({ok:false,error:"unknown"}); }catch(err){return out({ok:false,error:err.toString()})} } function out(obj){return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);}
